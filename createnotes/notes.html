            if rms < 1e-5:  # If no sound detected, stop recording and print the wave file name
                input_stream.close()
                return "wave_" + str(threshold) + ".wav"
    except Exception as e:
        print("Error:", e)

if __name__ == "__main__":

    print("Wave file name:", wave_file)
